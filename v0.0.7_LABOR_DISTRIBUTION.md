## 🎯 Especificación Funcional

### User Story

```gherkin
Feature: Labor Distribution

Scenario: Distribución proporcional al salario
  Given Una empresa con 2 empleados:
    | Nombre | Salario |
    | Juan   | $1,000  |
    | María  | $2,000  |
  
  And Un período con excedente de $900
  
  When Ejecuto distribución "proportional-salary"
  
  Then Juan recibe: $300 (33.33%)
  And María recibe: $600 (66.67%)
  
  And El total distribuido = $900
  And Cada cuota se puede auditar


Scenario: Distribución fija + variable
  Given Un período con:
    | Concepto | Monto |
    | Excedente | $1,000 |
    | Empleados | 2 |
  
  When Aplico regla "fixed-$500-plus-variable"
  
  Then Cada empleado recibe: $500 (fijo)
  And Sobrante $0 (variable proporcional a desempeño)
```

### 2. Servicio: LaborDistutionService

#### Archivo: `server/src/services/labor-distribution.service.js`

```javascript
import crypto from "crypto"
import logger from "../utils/logger.js"
import {
  ValidationError,
  InvalidSurplusError,
  ConflictError
} from "../utils/errors.js"

export default class LaborDistributionService {
  
  /**
   * Obtener regla de distribución activa de una empresa
   */
  async getActiveRule(companyId) {
    const rule = await DistributionRule.findOne({
      companyId,
      active: true
    })
    
    if (!rule) {
      // Usar regla por defecto
      return this.createDefaultRule(companyId)
    }
    
    return rule
  }
  
  /**
   * Crear regla por defecto: proporcional al salario
   */
  async createDefaultRule(companyId) {
    return await DistributionRule.create({
      companyId,
      name: "proportional-salary",
      description: "Distribución proporcional al salario base mensual",
      parameters: {},
      active: true,
      default: true
    })
  }
  
  /**
   * PRINCIPAL: Distribuir excedente según regla
   * 
   * @param {Object} period - Document de Period
   * @param {string} companyId - Para validar multi-tenant
   * @param {string} userId - Quién ejecuta
   * @returns {Object} Distribución {employeeId: amount}
   */
  async distributeExcedent(period, companyId, userId) {
    // Validaciones iniciales
    if (period.companyId.toString() !== companyId.toString()) {
      throw new Error("Unauthorized: Period belongs to different company")
    }
    
    if (period.distributionLocked) {
      throw new ConflictError("Distribution already locked (certified?)")
    }
    
    if (!period.result || period.result.surplus === undefined) {
      throw new ValidationError("Period must have calculated result with surplus")
    }
    
    const surplus = period.result.surplus
    
    if (surplus < 0) {
      return {
        distribution: {},  // Nada para distribuir
        message: "No surplus to distribute (negative result)",
        locked: true
      }
    }
    
    // Obtener regla
    const rule = await this.getActiveRule(companyId)
    
    // Obtener empleados
    const employees = await this.getEmployeesForPeriod(period, companyId)
    
    if (!employees.length) {
      throw new ValidationError("No employees found for distribution")
    }
    
    // Ejecutar distribución según regla
    let distribution
    
    switch (rule.name) {
      case "proportional-salary":
        distribution = this.distributeProportionalToSalary(
          surplus,
          employees
        )
        break
      
      case "proportional-productivity":
        distribution = this.distributeProportionalToProductivity(
          surplus,
          employees
        )
        break
      
      case "fixed-equal":
        distribution = this.distributeEqualFixed(
          surplus,
          employees
        )
        break
      
      case "fixed-plus-variable":
        distribution = this.distributeFixedPlusVariable(
          surplus,
          employees,
          rule.parameters
        )
        break
      
      case "custom":
        distribution = await this.distributeCustom(
          surplus,
          employees,
          rule.parameters
        )
        break
      
      default:
        throw new Error(`Unknown rule: ${rule.name}`)
    }
    
    // Auditoría: Crear hash de distribución
    const auditHash = this.createDistributionHash(distribution, rule)
    
    // Guardar en período
    const updatedPeriod = await Period.findByIdAndUpdate(
      period._id,
      {
        "distribution.byEmployee": distribution,
        "distribution.rule": {
          ruleId: rule._id,
          ruleName: rule.name,
          parameters: rule.parameters,
          version: rule.version || "v1"
        },
        "distribution.totalDistributed": 
          Object.values(distribution).reduce((sum, emp) => sum + (emp.total || 0), 0),
        "distribution.calculatedAt": new Date(),
        "distribution.calculatedBy": userId,
        "distribution.auditHash": auditHash
      },
      { new: true }
    )
    
    // Crear evento para auditoría
    await LaborEvent.create({
      periodId: period._id,
      companyId,
      type: "distribution-calculated",
      data: {
        rule: rule.name,
        totalDistributed: updatedPeriod.distribution.totalDistributed,
        employeeCount: employees.length
      },
      createdBy: userId,
      createdAt: new Date()
    })
    
    logger.info("LABOR", "Excedent distributed", {
      periodId: period._id,
      companyId,
      rule: rule.name,
      totalSurplus: surplus,
      employeeCount: employees.length,
      hash: auditHash
    })
    
    return updatedPeriod.distribution
  }
  
  /**
   * ALGORITMO 1: Proporcional al salario
   * 
   * Fórmula:
   *   Para cada empleado:
   *     share = (empleado.salary / totalSalaries) * surplus
   */
  distributeProportionalToSalary(surplus, employees) {
    const totalSalary = employees.reduce((sum, e) => sum + (e.baseSalary || 0), 0)
    
    if (totalSalary === 0) {
      throw new ValidationError("No employees with salary data")
    }
    
    const distribution = {}
    let totalDistributed = 0
    
    employees.forEach((emp) => {
      const ratio = emp.baseSalary / totalSalary
      const share = surplus * ratio
      
      distribution[emp._id.toString()] = {
        fixed: 0,
        variable: Math.floor(share * 100) / 100,  // 2 decimales
        total: Math.floor(share * 100) / 100,
        breakdown: {
          salaryRatio: ratio,
          productivityRatio: 0,
          salaryAmount: emp.baseSalary,
          calculatedShare: share
        }
      }
      
      totalDistributed += distribution[emp._id.toString()].total
    })
    
    // Manejo de redondeos: guardar diferencia
    const remainder = surplus - totalDistributed
    if (remainder > 0.01) {
      // Dar al empleado con mayor salario
      const topEmp = employees.reduce((max, e) => 
        e.baseSalary > (max?.baseSalary || 0) ? e : max
      )
      distribution[topEmp._id.toString()].total += remainder
      distribution[topEmp._id.toString()].variable += remainder
    }
    
    return distribution
  }
  
  /**
   * ALGORITMO 2: Proporcional a Productividad
   * Usa score de 0-100 asignado por manager
   */
  distributeProportionalToProductivity(surplus, employees) {
    const totalProductivity = employees.reduce((sum, e) => 
      sum + (e.productivityScore || 50), 0
    )
    
    const distribution = {}
    
    employees.forEach((emp) => {
      const score = emp.productivityScore || 50
      const ratio = score / totalProductivity
      const share = surplus * ratio
      
      distribution[emp._id.toString()] = {
        fixed: 0,
        variable: Math.floor(share * 100) / 100,
        total: Math.floor(share * 100) / 100,
        breakdown: {
          salaryRatio: 0,
          productivityRatio: ratio,
          productivityScore: score,
          calculatedShare: share
        }
      }
    })
    
    return distribution
  }
  
  /**
   * ALGORITMO 3: Fijo e igual para todos
   */
  distributeEqualFixed(surplus, employees) {
    const perEmployee = Math.floor((surplus / employees.length) * 100) / 100
    const distribution = {}
    
    employees.forEach((emp) => {
      distribution[emp._id.toString()] = {
        fixed: perEmployee,
        variable: 0,
        total: perEmployee,
        breakdown: {
          fixedAmount: perEmployee,
          employeeCount: employees.length
        }
      }
    })
    
    return distribution
  }
  
  /**
   * ALGORITMO 4: Fijo ($X) + Variable (resto proporcional a métrica)
   */
  distributeFixedPlusVariable(surplus, employees, params) {
    const { fixedAmount = 100, variableMetric = "salary" } = params
    
    const totalFixed = fixedAmount * employees.length
    const remaining = surplus - totalFixed
    
    if (remaining < 0) {
      logger.warn("LABOR", "Surplus less than fixed amount for all employees")
      // Distribuir lo que hay
      const perEmployee = Math.floor((surplus / employees.length) * 100) / 100
      return employees.reduce((dist, emp) => ({
        ...dist,
        [emp._id.toString()]: {
          fixed: perEmployee,
          variable: 0,
          total: perEmployee
        }
      }), {})
    }
    
    // Calcular variable según métrica
    const distribution = {}
    
    if (variableMetric === "salary") {
      const totalSalary = employees.reduce((sum, e) => sum + (e.baseSalary || 0), 0)
      
      employees.forEach((emp) => {
        const ratio = emp.baseSalary / totalSalary
        const variable = remaining * ratio
        
        distribution[emp._id.toString()] = {
          fixed: fixedAmount,
          variable: Math.floor(variable * 100) / 100,
          total: fixedAmount + Math.floor(variable * 100) / 100
        }
      })
    }
    
    return distribution
  }
  
  /**
   * Algoritmo personalizado: Ejecuta fórmula custom
   */
  async distributeCustom(surplus, employees, params) {
    const { formula } = params
    
    // Ejecutar fórmula en contexto sandboxed
    const vm = require("vm")
    const sandbox = {
      surplus,
      employees: employees.map(e => ({
        id: e._id.toString(),
        salary: e.baseSalary,
        productivity: e.productivityScore || 50,
        hours: e.hoursWorked || 160
      }))
    }
    
    const script = new vm.Script(`
      const result = {};
      ${formula}
      result;
    `)
    
    try {
      const distribution = script.runInNewContext(sandbox, {
        timeout: 5000
      })
      
      return distribution
    } catch (error) {
      throw new ValidationError(`Custom formula error: ${error.message}`)
    }
  }
  
  /**
   * Obtener empleados y su data para un período
   */
  async getEmployeesForPeriod(period, companyId) {
    const employees = await Employee.find({
      companyId,
      status: { $in: ["active", "on-leave"] }
    }).select("baseSalary productivityScore hoursWorked")
    
    return employees
  }
  
  /**
   * Crear hash SHA256 para auditoría
   */
  createDistributionHash(distribution, rule) {
    const data = {
      distribution,
      rule: rule.name,
      timestamp: Math.floor(Date.now() / 1000)  // Segundo, no milisegundo
    }
    
    return crypto
      .createHash("sha256")
      .update(JSON.stringify(data))
      .digest("hex")
  }
  
  /**
   * API: Obtener distribución de un empleado (su share)
   */
  async getEmployeeShare(periodId, employeeId, companyId) {
    const period = await Period.findOne({
      _id: periodId,
      companyId
    })
    
    if (!period || !period.distribution?.byEmployee) {
      throw new Error("Period has no distribution")
    }
    
    const employee = await Employee.findOne({
      _id: employeeId,
      companyId
    })
    
    if (!employee) {
      throw new Error("Employee not found")
    }
    
    const share = period.distribution.byEmployee[employeeId]
    
    return {
      period: {
        month: period.month,
        year: period.year,
        surplus: period.result?.surplus
      },
      employee: {
        id: employee._id,
        name: employee.name,
        email: employee.email,
        baseSalary: employee.baseSalary,
        productivityScore: employee.productivityScore
      },
      calculation: {
        rule: period.distribution.rule.ruleName,
        breakdown: share.breakdown
      },
      distribution: {
        fixed: share.fixed,
        variable: share.variable,
        total: share.total,
        currency: "ARS"
      },
      auditTrail: {
        calculatedAt: period.distribution.calculatedAt,
        calculatedBy: period.distribution.calculatedBy,
        auditHash: period.distribution.auditHash
      }
    }
  }
}

export default new LaborDistributionService()
```

```javascript
import laborService from "../../services/labor-distribution.service.js"
import { asyncHandler } from "../../middleware/errorHandler.js"
import { sendSuccess, sendError } from "../../utils/response.js"

/**
 * POST /api/labor/distribute/:periodId
 * Distribuir excedente según regla activa de la empresa
 */
export const distributePeriod = asyncHandler(async (req, res) => {
  const { periodId } = req.params
  const { companyId, userId } = req
  
  if (!companyId) {
    return sendError(res, 403, "Company context required")
  }
  
  const period = await Period.findOne({
    _id: periodId,
    companyId
  })
  
  if (!period) {
    return sendError(res, 404, "Period not found")
  }
  
  const distribution = await laborService.distributeExcedent(
    period,
    companyId,
    userId
  )
  
  sendSuccess(res, distribution, "Distribution calculated successfully")
})

/**
 * GET /api/labor/my-share/:periodId
 * Obtener la distribución personal (empleado ve su cuota)
 */
export const getMyShare = asyncHandler(async (req, res) => {
  const { periodId } = req.params
  const { userId, companyId } = req
  
  // Encontrar el employee asociado a este user
  const employee = await Employee.findOne({
    userId,
    companyId
  })
  
  if (!employee) {
    return sendError(res, 404, "Employee record not found")
  }
  
  const share = await laborService.getEmployeeShare(
    periodId,
    employee._id,
    companyId
  )
  
  sendSuccess(res, share)
})

/**
 * GET /api/labor/rules
 * Listar reglas de distribución activas
 */
export const listRules = asyncHandler(async (req, res) => {
  const { companyId } = req
  
  const rules = await DistributionRule.find({
    companyId,
    active: true
  }).select('-customFormula')
  
  sendSuccess(res, rules)
})

/**
 * POST /api/labor/rules
 * Crear nueva regla de distribución
 */
export const createRule = asyncHandler(async (req, res) => {
  const { companyId, userId } = req
  const { name, description, parameters } = req.body
  
  // Validar que sea admin
  const user = await User.findById(userId)
  if (user.role !== "admin" && user.role !== "company") {
    return sendError(res, 403, "Only admin/company can create rules")
  }
  
  const rule = await DistributionRule.create({
    companyId,
    name,
    description,
    parameters,
    createdBy: userId,
    active: true
  })
  
  sendSuccess(res, rule, "Rule created", 201)
})
```
