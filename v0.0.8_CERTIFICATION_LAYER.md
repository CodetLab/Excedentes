```gherkin
Feature: Period Certification & Locking

Scenario: Certificar un período cierra cambios
  Given Un período de enero 2026 con:
    | Campo | Valor |
    | Ventas | $5,000 |
    | Costos | $2,000 |
    | Resultado (excedente) | $3,000 |
    | Distribución | [Juan: $1000, María: $2000] |
  
  When Doy click en "CERTIFICAR PERÍODO"
  
  Then Se genera:
    ✓ Certificate con SHA256(inputs + outputs)
    ✓ Período marcado como locked: true
    ✓ Período marcado como certified: true
    ✓ Descarga PDF de certificado
  
  And Si alguien intenta MODIFICAR:
    ✓ Rechazado: "Period is locked"
    ✓ Se genera evento de "intento de manipulación"
    ✓ Se alerta a admin


Scenario: Verificar integridad del certificado
  Given Un período certificado con hash SHA256
  
  When Ejecuto: GET /api/certifications/verify/:periodId
  
  Then Obtengo:
    ✓ Certificate válido: true/false
    ✓ Inputs hash coincide: true/false
    ✓ Outputs hash coincide: true/false
    ✓ Detalles: fecha emisión, signado por, fecha expiración
```

### 2. Servicio: CertificationService

#### Archivo: `server/src/services/certification.service.js`

```javascript
import crypto from "crypto"
import logger from "../utils/logger.js"
import Certificate from "../models/Certificate.js"
import ComplianceAudit from "../models/ComplianceAudit.js"
import Period from "../models/Period.js"

const CERTIFICATE_VALIDITY_YEARS = 7
const HASH_ALGORITHM = "sha256"

export default class CertificationService {
  
  /**
   * Certificar un período (punto de no retorno)
   * 
   * Pasos:
   * 1. Verificar que período esté completo (result + distribution)
   * 2. Crear snapshots de inputs y outputs
   * 3. Calcular hashes SHA256
   * 4. Crear Certificate document
   * 5. LOCK el período (locked: true, certified: true)
   * 6. Retornar certificado con datos para PDF
   */
  async certificatePeriod(periodId, companyId, userId) {
    // 1. Validar período
    const period = await Period.findOne({
      _id: periodId,
      companyId
    })
    
    if (!period) {
      throw new Error("Period not found")
    }
    
    if (period.certified) {
      throw new Error("Period already certified")
    }
    
    if (period.locked) {
      throw new Error("Period is locked (cannot modify or certify)")
    }
    
    if (!period.result || !period.distribution) {
      throw new Error("Period must have result and distribution calculated")
    }
    
    // 2. Crear snapshots de datos
    const inputsSnapshot = this.createInputSnapshot(period)
    const outputsSnapshot = this.createOutputSnapshot(period)
    
    // 3. Calcular hashes
    const inputsHash = this.calculateHash(inputsSnapshot)
    const outputsHash = this.calculateHash(outputsSnapshot)
    
    // 4. Crear documento de Certificate
    const certificateNumber = await this.generateCertificateNumber(companyId)
    
    const certificate = new Certificate({
      periodId: period._id,
      companyId,
      
      inputsSnapshot,
      inputsHash,
      
      outputsSnapshot,
      outputsHash,
      
      certificateNumber,
      issuedAt: new Date(),
      issuedBy: userId,
      expiresAt: new Date(Date.now() + CERTIFICATE_VALIDITY_YEARS * 365 * 24 * 60 * 60 * 1000),
      
      legalFramework: {
        country: "AR",
        legislation: "Ley de Contrato de Trabajo 20744",
        complianceTests: []
      },
      
      isValid: true
    })
    
    await certificate.save()
    
    logger.info("CERT", "Certificate created", {
      periodId: period._id,
      companyId,
      certificateNumber,
      inputsHash: inputsHash.substring(0, 16) + "...",
      outputsHash: outputsHash.substring(0, 16) + "..."
    })
    
    // 5. LOCK el período
    const lockedPeriod = await Period.findByIdAndUpdate(
      periodId,
      {
        certified: true,
        locked: true,
        certificateId: certificate._id,
        lockedAt: new Date(),
        lockedReason: "certified",
        status: "certified"  // Cambiar estado a certified
      },
      { new: true }
    )
    
    logger.warn("CERT", "Period locked after certification", {
      periodId: period._id,
      month: period.month,
      year: period.year
    })
    
    // 6. Retornar con datos para PDF
    return {
      certificate: {
        id: certificate._id,
        number: certificateNumber,
        issuedAt: certificate.issuedAt,
        expiresAt: certificate.expiresAt,
        inputsHash,
        outputsHash,
        verificationUrl: `/api/certifications/verify/${periodId}`
      },
      period: {
        month: lockedPeriod.month,
        year: lockedPeriod.year,
        surplus: lockedPeriod.result.surplus,
        totalDistributed: lockedPeriod.distribution.totalDistributed
      }
    }
  }
  
  /**
   * Crear snapshot de INPUTS (datos que entraron)
   */
  createInputSnapshot(period) {
    return {
      month: period.month,
      year: period.year,
      
      // Ventas consolidadas
      sales: {
        byProduct: period.sales?.byProduct || {},
        total: period.sales?.total || 0,
        recordCount: period.sales?.recordCount || 0,
        lastUpdatedAt: period.sales?.lastUpdatedAt
      },
      
      // Costos consolidados
      costs: {
        byCategory: period.costs?.byCategory || {},
        total: period.costs?.total || 0,
        recordCount: period.costs?.recordCount || 0,
        lastUpdatedAt: period.costs?.lastUpdatedAt
      },
      
      // Datos de empleados
      employees: {
        count: period.employees?.count || 0,
        salaryPool: period.employees?.salaryPool || 0,
        activeCount: period.employees?.activeCount || 0
      },
      
      // Gastos extras
      extraExpenses: {
        items: period.extraExpenses?.items || [],
        total: period.extraExpenses?.total || 0
      },
      
      // Meta: cuándo se procesaron los datos
      processedAt: period.calculatedAt,
      processedBy: period.calculatedBy
    }
  }
  
  /**
   * Crear snapshot de OUTPUTS (resultados del cálculo)
   */
  createOutputSnapshot(period) {
    return {
      // Resultado económico
      result: {
        revenues: period.result.revenues,
        expenses: period.result.expenses,
        surplus: period.result.surplus,
        salaryCosts: period.result.salaryCosts,
        overheadCosts: period.result.overheadCosts,
        otherCosts: period.result.otherCosts
      },
      
      // Distribución
      distribution: {
        rule: period.distribution.rule,
        byEmployee: period.distribution.byEmployee,
        totalDistributed: period.distribution.totalDistributed,
        remainingBuffer: period.distribution.remainingBuffer
      },
      
      // Status
      certificationTimestamp: new Date(),
      dataIntegrity: {
        totalInputRecords: 
          (period.sales?.recordCount || 0) +
          (period.costs?.recordCount || 0),
        checksumValid: true,
        anomaliesDetected: false
      }
    }
  }
  
  /**
   * Calcular SHA256 de un objeto
   */
  calculateHash(data) {
    // Serializar con orden determinístico
    const serialized = JSON.stringify(data, Object.keys(data).sort())
    
    return crypto
      .createHash(HASH_ALGORITHM)
      .update(serialized)
      .digest("hex")
  }
  
  /**
   * Generar número único de certificado
   * Formato: CERT-AAAA-MM-NNN
   */
  async generateCertificateNumber(companyId) {
    const now = new Date()
    const year = now.getFullYear()
    const month = String(now.getMonth() + 1).padStart(2, "0")
    
    // Encontrar último número del mes/año
    const lastCert = await Certificate.findOne(
      { companyId },
      {},
      { sort: { createdAt: -1 } }
    )
    
    let sequence = 1
    if (lastCert?.certificateNumber) {
      const lastNumber = lastCert.certificateNumber // "CERT-2026-02-001"
      const lastSeq = parseInt(lastNumber.split("-")[3]) || 0
      sequence = lastSeq + 1
    }
    
    return `CERT-${year}-${month}-${String(sequence).padStart(3, "0")}`
  }
  
  /**
   * VERIFICACIÓN: Validar que un certificado no fue manipulado
   * 
   * Steps:
   * 1. Obtener datos actuales del período
   * 2. Recalcular hashes
   * 3. Comparar con hashes almacenados
   * 4. Retornar resultado de verificación
   */
  async verifyCertificate(periodId, companyId) {
    const certificate = await Certificate.findOne({
      periodId,
      companyId
    })
    
    if (!certificate) {
      return {
        valid: false,
        reason: "Certificate not found"
      }
    }
    
    const period = await Period.findById(periodId)
    
    if (!period) {
      return {
        valid: false,
        reason: "Period not found"
      }
    }
    
    // Recalcular inputs hash
    const currentInputsSnapshot = this.createInputSnapshot(period)
    const currentInputsHash = this.calculateHash(currentInputsSnapshot)
    
    // Recalcular outputs hash
    const currentOutputsSnapshot = this.createOutputSnapshot(period)
    const currentOutputsHash = this.calculateHash(currentOutputsSnapshot)
    
    // Comparar
    const inputsMatchInputHash = currentInputsHash === certificate.inputsHash
    const outputsMatchOutputsHash = currentOutputsHash === certificate.outputsHash
    
    const certificateIsValid = inputsMatchInputHash && outputsMatchOutputsHash
    
    // Si detectamos manipulación, marcar y alertar
    if (!certificateIsValid && certificate.isValid === true) {
      await Certificate.findByIdAndUpdate(certificate._id, {
        isValid: false,
        tamperedAt: new Date(),
        tamperedReason: `Hash mismatch: inputs=${inputsMatchInputHash}, outputs=${outputsMatchOutputsHash}`
      })
      
      logger.error("SECURITY", "CERTIFICATE TAMPERING DETECTED", {
        periodId,
        companyId,
        certificateNumber: certificate.certificateNumber,
        inputsMatch: inputsMatchInputHash,
        outputsMatch: outputsMatchOutputsHash
      })
    }
    
    return {
      valid: certificateIsValid,
      certificate: {
        number: certificate.certificateNumber,
        issuedAt: certificate.issuedAt,
        expiresAt: certificate.expiresAt,
        isValid: certificate.isValid
      },
      verification: {
        inputsMatch: inputsMatchInputHash,
        outputsMatch: outputsMatchOutputsHash,
        integrity: certificateIsValid ? "VALID" : "TAMPERED"
      },
      period: {
        month: period.month,
        year: period.year,
        status: period.status,
        locked: period.locked
      }
    }
  }
  
  /**
   * Obtener certificado de un período
   */
  async getCertificate(periodId, companyId) {
    const certificate = await Certificate.findOne({
      periodId,
      companyId
    }).populate("issuedBy", "name email")
    
    if (!certificate) {
      throw new Error("Certificate not found")
    }
    
    return certificate
  }
  
  /**
   * Listar certificados de una empresa
   */
  async listCertificates(companyId, options = {}) {
    const { skip = 0, limit = 20, sort = { issuedAt: -1 } } = options
    
    const [certificates, total] = await Promise.all([
      Certificate.find({ companyId })
        .sort(sort)
        .skip(skip)
        .limit(limit)
        .populate("periodId", "month year")
        .populate("issuedBy", "name email"),
      Certificate.countDocuments({ companyId })
    ])
    
    return { certificates, total, skip, limit }
  }
  
  /**
   * Pre-certification audit
   * Validar que período esté listo para certificación
   */
  async runPreCertificationAudit(periodId, companyId) {
    const period = await Period.findOne({
      _id: periodId,
      companyId
    })
    
    if (!period) {
      throw new Error("Period not found")
    }
    
    const tests = []
    let allPassed = true
    
    // Test 1: Result calculated
    tests.push({
      name: "result-calculated",
      description: "Período tiene resultado económico calculado",
      passed: !!period.result && typeof period.result.surplus === "number",
      details: { surplus: period.result?.surplus }
    })
    
    // Test 2: Distribution calculated
    tests.push({
      name: "distribution-calculated",
      description: "Distribución de excedentes calculada",
      passed: !!period.distribution?.byEmployee,
      details: {
        employeeCount: Object.keys(period.distribution?.byEmployee || {}).length
      }
    })
    
    // Test 3: No negative surplus
    tests.push({
      name: "no-negative-surplus",
      description: "Excedente no es negativo",
      passed: (period.result?.surplus || 0) >= 0,
      details: { surplus: period.result?.surplus }
    })
    
    // Test 4: Distribution sum equals surplus
    if (period.distribution?.byEmployee) {
      const totalDist = Object.values(period.distribution.byEmployee)
        .reduce((sum, emp) => sum + (emp.total || 0), 0)
      
      const match = Math.abs(totalDist - (period.result?.surplus || 0)) < 0.01
      
      tests.push({
        name: "distribution-sum-accuracy",
        description: "Suma de distribución = excedente",
        passed: match,
        details: {
          totalSurplus: period.result?.surplus,
          totalDistributed: totalDist,
          difference: period.result?.surplus - totalDist
        }
      })
    }
    
    // Test 5: All required employees have distribution
    const employees = await Employee.countDocuments({
      companyId,
      status: { $in: ["active", "on-leave"] }
    })
    
    const distributedTo = Object.keys(period.distribution?.byEmployee || {}).length
    
    tests.push({
      name: "all-employees-covered",
      description: "Todos los empleados activos en distribución",
      passed: distributedTo === employees,
      details: { activeEmployees: employees, distributed: distributedTo }
    })
    
    // Determinar resultado
    allPassed = tests.every(t => t.passed)
    
    const audit = await ComplianceAudit.create({
      periodId,
      companyId,
      auditType: "pre-certification",
      tests,
      summary: allPassed 
        ? "Period ready for certification" 
        : "Period has issues, cannot certify",
      riskLevel: allPassed ? "LOW" : "HIGH",
      recommendations: allPassed ? [] : [
        "Fix failed tests before certification",
        "Review calculation logs",
        "Contact support if issues persist"
      ],
      executedAt: new Date()
    })
    
    return {
      auditId: audit._id,
      passed: allPassed,
      tests,
      recommendations: audit.recommendations
    }
  }
}

export default new CertificationService()
```

### 3. Controlador: CertificationController

```javascript
// server/src/api/controllers/certification.controller.js

import certService from "../../services/certification.service.js"
import { asyncHandler } from "../../middleware/errorHandler.js"
import { sendSuccess, sendError } from "../../utils/response.js"

/**
 * POST /api/certifications/certify/:periodId
 * Certificar un período (PUNTO DE NO RETORNO)
 */
export const certifyPeriod = asyncHandler(async (req, res) => {
  const { periodId } = req.params
  const { companyId, userId } = req
  
  if (!companyId) {
    return sendError(res, 403, "Company context required")
  }
  
  // Pre-check: audit
  const auditResult = await certService.runPreCertificationAudit(periodId, companyId)
  
  if (!auditResult.passed) {
    return sendError(res, 400, "Period failed pre-certification audit", {
      audit: auditResult
    })
  }
  
  // Certificar
  const result = await certService.certificatePeriod(periodId, companyId, userId)
  
  sendSuccess(res, result, "Period certified successfully")
})

/**
 * GET /api/certifications/verify/:periodId
 * Verificar la integridad de un certificado
 */
export const verifyCertificate = asyncHandler(async (req, res) => {
  const { periodId } = req.params
  const { companyId } = req
  
  const result = await certService.verifyCertificate(periodId, companyId)
  
  sendSuccess(res, result)
})

/**
 * GET /api/certifications/:periodId
 * Obtener certificado de un período
 */
export const getCertificate = asyncHandler(async (req, res) => {
  const { periodId } = req.params
  const { companyId } = req
  
  const certificate = await certService.getCertificate(periodId, companyId)
  
  sendSuccess(res, certificate)
})

/**
 * GET /api/certifications
 * Listar todos los certificados de una empresa
 */
export const listCertificates = asyncHandler(async (req, res) => {
  const { companyId } = req
  const { skip = 0, limit = 20 } = req.query
  
  const result = await certService.listCertificates(companyId, {
    skip: parseInt(skip),
    limit: parseInt(limit)
  })
  
  sendSuccess(res, result)
})

/**
 * GET /api/certifications/audit/:periodId
 * Obtener resultado del audit pre-certificación
 */
export const getCertificationAudit = asyncHandler(async (req, res) => {
  const { periodId } = req.params
  const { companyId } = req
  
  const audit = await certService.runPreCertificationAudit(periodId, companyId)
  
  sendSuccess(res, audit)
```

```typescript
// client/src/components/CertificationModal.tsx

import React, { useState } from "react"
import apiClient from "../services/apiClient"
import { Modal, Button, Alert } from "./index"

interface CertificationModalProps {
  periodId: string
  month: number
  year: number
  onSuccess?: (certificate: any) => void
}

export const CertificationModal: React.FC<CertificationModalProps> = ({
  periodId,
  month,
  year,
  onSuccess
}) => {
  const [isOpen, setIsOpen] = useState(false)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [auditResults, setAuditResults] = useState<any>(null)
  const [step, setStep] = useState<"audit" | "confirm" | "done">("audit")
  const [certificate, setCertificate] = useState<any>(null)

    setLoading(true)
    setError(null)
    
    try {
      const response = await apiClient.get(
        `/api/certifications/audit/${periodId}`
      )
      setAuditResults(response.data.data)
    } catch (err) {
      setError(err instanceof Error ? err.message : "Audit failed")

      setLoading(false)
    }

    setLoading(true)

      const response = await apiClient.post(
        `/api/certifications/certify/${periodId}`,
        {}
      )
      setCertificate(response.data.data.certificate)
      setStep("done")
      
      if (onSuccess) {
        onSuccess(response.data.data)
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : "Certification failed")
    } finally {
      setLoading(false)
    }
  }
  
  return (
    <>
      <Button
        onClick={() => {
          setIsOpen(true)
          setStep("audit")
          runAuditCheck()
        }}
        className="btn-certify"
      >
        🔐 Certificar Período
      </Button>
      
      {isOpen && (
        <Modal title="Certificar Período" onClose={() => setIsOpen(false)}>
          {error && <Alert type="error">{error}</Alert>}
          
          {step === "audit" && (
            <div>
              <p>Verificando que el período esté listo para certificación...</p>
              
              {loading && <p>Ejecutando auditoría...</p>}
              
              {auditResults && (
                <div>
                  <div className="audit-results">
                    {auditResults.tests.map((test: any) => (
                      <div
                        key={test.name}
                        className={`test ${test.passed ? "passed" : "failed"}`}
                      >
                        <span className="checkbox">
                          {test.passed ? "✓" : "✗"}
                        </span>
                        <span>{test.description}</span>
                      </div>
                    ))}
                  </div>
                  
                  {auditResults.passed ? (
                    <div className="buttons">
                      <Button
                        onClick={() => setStep("confirm")}
                        className="btn-primary"
                      >
                        Continuar a Certificación
                      </Button>
                      <Button
                        onClick={() => setIsOpen(false)}
                        className="btn-secondary"
                      >
                        Cancelar
                      </Button>
                    </div>
                  ) : (
                    <Alert type="warning">
                      El período no está listo. Revisa los errores y vuelve a intentar.
                    </Alert>
                  )}
                </div>
              )}
            </div>
          )}
          
          {step === "confirm" && (
            <div>
              <Alert type="warning">
                <strong>⚠️ ATENCIÓN:</strong> Una vez certificado, este período no podrá ser modificado.
              </Alert>
              
              <p>
                Se va a certificar el período <strong>{month}/{year}</strong> con el siguiente estado:
              </p>
              
              <div className="summary">
                <p>✓ Resultado económico calculado</p>
                <p>✓ Distribución para empleados realizada</p>
                <p>✓ Auditoría pre-certificación: APROBADA</p>
              </div>
              
              <div className="buttons">
                <Button
                  onClick={confirmCertification}
                  disabled={loading}
                  className="btn-danger"
                >
                  {loading ? "Certificando..." : "SÍ, CERTIFICAR AHORA"}
                </Button>
                <Button
                  onClick={() => setStep("audit")}
                  disabled={loading}
                  className="btn-secondary"
                >
                  Volver
                </Button>
              </div>
            </div>
          )}
          
          {step === "done" && certificate && (
            <div className="certification-done">
              <div className="success-icon">✓</div>
              
              <h3>¡Período Certificado!</h3>
              
              <div className="certificate-info">
                <p>
                  <strong>Número de Certificado:</strong> {certificate.number}
                </p>
                <p>
                  <strong>Emitido:</strong>{" "}
                  {new Date(certificate.issuedAt).toLocaleDateString("es-AR")}
                </p>
                <p>
                  <strong>Válido hasta:</strong>{" "}
                  {new Date(certificate.expiresAt).toLocaleDateString("es-AR")}
                </p>
                
                <div className="hash-section">
                  <p>
                    <strong>Hash de Entrada:</strong>
                    <code>{certificate.inputsHash.substring(0, 32)}...</code>
                  </p>
                  <p>
                    <strong>Hash de Salida:</strong>
                    <code>{certificate.outputsHash.substring(0, 32)}...</code>
                  </p>
                </div>
              </div>
              
              <div className="buttons">
                <Button onClick={() => setIsOpen(false)} className="btn-primary">
                  Cerrar
                </Button>
              </div>
            </div>
          )}
        </Modal>
      )}
    </>
  )
}
```
